STEP 3: REPLACE TRAILING STOP CODE IN ManageOpenPositions() FUNCTION
-------------------------------------------------------------------

Replace the existing trailing stop code with this (keep everything else the same):

```
      // Enhanced multi-stage trailing stop with progressive tightening
      if(profit_ratio >= TRAILING_TRIGGER && USE_MULTI_STAGE_TRAILING)
      {
         double new_sl = current_sl;
         double trailing_step = TRAILING_STEP;
         
         // Stage 1: Initial breakeven (50% of stop distance) - Earlier than before
         if(profit_ratio >= 0.5 && profit_ratio < 1.0)
         {
            if(position_type == POSITION_TYPE_BUY)
               new_sl = open_price + 5 * _Point; // Small buffer in profit
            else
               new_sl = open_price - 5 * _Point;
               
            if(VERBOSE_MODE)
               Print("Moving to breakeven+ at profit ratio: ", DoubleToString(profit_ratio, 2));
         }
         // Stage 2: Standard trailing (100% of stop distance)
         else if(profit_ratio >= 1.0 && profit_ratio < 1.5)
         {
            if(position_type == POSITION_TYPE_BUY)
               new_sl = current_price - trailing_step * _Point;
            else
               new_sl = current_price + trailing_step * _Point;
               
            if(VERBOSE_MODE)
               Print("Applying standard trailing at profit ratio: ", DoubleToString(profit_ratio, 2));
         }
         // Stage 3: Tighter trailing (150% of stop distance)
         else if(profit_ratio >= 1.5)
         {
            // Use tighter trailing for larger profits (70% of standard)
            trailing_step = trailing_step * 0.7;
            
            if(position_type == POSITION_TYPE_BUY)
               new_sl = current_price - trailing_step * _Point;
            else
               new_sl = current_price + trailing_step * _Point;
               
            if(VERBOSE_MODE)
               Print("Applying tighter trailing at profit ratio: ", DoubleToString(profit_ratio, 2));
         }
         
         // Only modify if the new stop loss is better than current
         if((position_type == POSITION_TYPE_BUY && new_sl > stop_loss) ||
            (position_type == POSITION_TYPE_SELL && (new_sl < stop_loss || stop_loss == 0)))
         {
            new_sl = NormalizeDouble(new_sl, _Digits);
            
            if(trade.PositionModify(ticket, new_sl, take_profit))
            {
               if(VERBOSE_MODE)
                  Print("Applied multi-stage trailing stop: ", DoubleToString(new_sl, _Digits), 
                        " (Profit ratio: ", DoubleToString(profit_ratio, 2), ")");
            }
         }
      }
      // Standard trailing if multi-stage is disabled
      else if(profit_ratio >= TRAILING_TRIGGER && !USE_MULTI_STAGE_TRAILING)
      {
         // Your existing trailing stop code here
      }
```

STEP 5: ADD SMARTER PARTIAL CLOSE IN ManageOpenPositions()
--------------------------------------------------------

Find where you handle partial closes and enhance it like this:

```
      // Progressive partial close strategy based on profit ratio
      if(USE_PARTIAL_CLOSE && profit_ratio >= PARTIAL_CLOSE_TRIGGER)
      {
         // Check if already partially closed
         bool already_partially_closed = false;
         for(int j = 0; j < ArraySize(g_partially_closed_tickets); j++)
         {
            if(g_partially_closed_tickets[j] == ticket)
            {
               already_partially_closed = true;
               break;
            }
         }
         
         if(!already_partially_closed)
         {
            // Calculate partial close percentage based on profit ratio
            double partial_percent = PARTIAL_CLOSE_PERCENT;
            
            // Increase partial close percentage for larger profits
            if(profit_ratio >= 1.5) // Significantly in profit
               partial_percent = 60.0; // Close 60%
            else if(profit_ratio >= 1.2) // Moderately in profit
               partial_percent = 50.0; // Close 50%
            
            double partial_volume = position_volume * partial_percent / 100.0;
            
            // Ensure partial volume meets minimum lot size
            double min_lot = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN);
            double remaining_volume = position_volume - partial_volume;
            
            if(partial_volume >= min_lot && remaining_volume >= min_lot)
            {
               if(trade.PositionClosePartial(ticket, partial_volume))
               {
                  // Add to partially closed tickets array
                  int size = ArraySize(g_partially_closed_tickets);
                  ArrayResize(g_partially_closed_tickets, size + 1);
                  g_partially_closed_tickets[size] = ticket;
                  
                  if(VERBOSE_MODE)
                     Print("Partially closed position ", ticket, " (", 
                           DoubleToString(partial_percent, 0), "%)");
               }
            }
         }
      }
```

STEP 6: ADD STAGNANT TRADE CLOSURE (if not already present)
---------------------------------------------------------

Add this code to ManageOpenPositions() to close stagnant trades:

```
      // Close stagnant positions after 8 hours if in loss
      datetime position_time = (datetime)PositionGetInteger(POSITION_TIME);
      int position_age_hours = (int)((TimeCurrent() - position_time) / 3600);
      
      if(position_age_hours >= 8 && position_profit < 0) // Position open more than 8 hours and in loss
      {
         if(MathAbs(position_profit) > 0.5 * (MathAbs(stop_loss - open_price) * position_volume * 10))
         {
            // If position is losing more than half of expected stop loss value
            if(trade.PositionClose(ticket))
            {
               if(VERBOSE_MODE)
                  Print("Closed stagnant losing position ", ticket, " after ", position_age_hours, " hours");
            }
            continue;
         }
      }
```
